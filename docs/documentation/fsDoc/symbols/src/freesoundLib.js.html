 11 /**
 12  * @class FS is the namespace of the freesound-javascript library 
 13  */
 14 var FS = new function()
 15 {
 16 		
 17 	this._URI_SOUND = '/sounds/<sound_id>/';
 18 	this._URI_SOUND_ANALYSIS = '/sounds/<sound_id>/analysis/<filter>/';
 19 	this._URI_SIMILAR_SOUNDS = '/sounds/<sound_id>/similar/';
 20 	this._URI_SEARCH = '/sounds/search/';
 21 	this._URI_USER = '/people/<user_name>/';
 22 	this._URI_USER_SOUNDS = '/people/<user_name>/sounds/';
 23 	this._URI_USER_PACKS = '/people/<user_name>/packs/';
 24 	this._URI_PACK = '/packs/<pack_id>/';
 25 	this._URI_PACK_SOUNDS = '/packs/<pack_id>/sounds/';
 26 	
 27  	/**
 28  	 * Method to create uri for request. 
 29  	 * @param {string} uri Contains the standerd string for request. 
 30  	 * @param {array} args Contains the arguments that needs to be placed in standard string.
 31  	 * @returns {string} uri Contains the total uri for the request.
 32  	 */
 33 	this._uri = function(uri, args){
 34 		for (a in args) {
 35 			uri = uri.replace(/<[\w_]+>/, args[a]);
 36 		};
 37 		uri = FS.FreesoundData.getBaseUri() + uri;
 38 		return uri;
 39 	};
 40 	
 41 	/**
 42 	 * Method to split reference uri in uri and paramters
 43 	 * @param {string} uri Contains the uri to split in uri and parameters 
 44 	 */
 45 	this._splitUri = function(uri)
 46 	{
 47 		//test uri for ? -> contains parameters
 48 		if(/\?/.test(uri)){
 49 			//split up uri
 50 			var splittedUri = uri.split(/\?/);
 51 			//save baseUri
 52 			var baseUri = splittedUri[0];
 53 			//save parameters
 54 			var parameters = splittedUri[1];
 55 			//test params for & -> contains more then one parameter
 56 			if(/\&/.test(parameters)){	
 57 				//split parameters
 58 				var params =  parameters.split(/\&/); 
 59 			}
 60 			//else safe single parameter in array to be able to create parameter object
 61 			else{var params = [parameters]; };
 62 			//create object to place parameters in
 63 			var parametersObject = {}; 
 64 			//split up parameters strings and place in object
 65 			for( a in params)
 66 			{	
 67 				var param = params[a].split(/\=/);
 68 				eval("parametersObject."+ param[0] + "=\"" + param[1] + "\"");
 69 			}
 70 			//return array with baseUri and paramtersObject
 71 			return [baseUri, parametersObject];
 72 		} else{
 73 			return [uri, false];	
 74 		}
 75 	}
 76 }

169 /**
170  * Standard error function, used whith json request when no error callback method is passed.
171  * @static
172  * @param {object} XMLHttpRequest
173  * @param {string} textStatus Describes the type of error that occurred.
174  * @param {object} errorThrown Optional exception object.
175  */
176 FS.RequestCreator.standardErrorMethod = function(XMLHttpRequest, textStatus, errorThrown){	
177 	var errorProps = eval('(' + XMLHttpRequest.responseText + ')');
178 	throw new Error("RequestCreator error, request didn't succeed. " + errorProps['explanation']);
179 };
180 
181 /**
182  * Create a request to server. 
183  * @static
184  * @param {string} uri Holds the uri for the request.
185  * @param {function} succesCallback Will be called when the request succeeds.
186  * @param {fucntion} errorCallback Will be called when the request fails
187  * (no XMLHttpRequest and specified error passed to errorCallback with use of jsonp)
188  * @param {array} params Contains the parameters as String that need to be send with the request.
189  */
190 FS.RequestCreator.createGetReq = function(uri, succesCallback, errorCallback, params){	
191 	//get api key, needed to create a request	
192 	var aKey = FS.FreesoundData.getApiKey();
193 	//create parameter object, with api key and passed paramameters
194 	var dataParams = {api_key: aKey};
195 	//if params are passed, extend dataParams with params
196 	if (params) { $.extend(dataParams, params); };
197 	//check if errorCallback is a function, 
198 	//if not -> use standard error method, else combine standard errorCallback with passed errorcallback
199 		
200 	if(!$.isFunction(errorCallback)){	
201 		var newErrorCallback = FS.RequestCreator.standardErrorMethod;
202 	} else {
203 		var newErrorCallback = function(XMLHttpRequest, textStatus, errorThrown){
204 			//FS.RequestCreator.standardErrorMethod(XMLHttpRequest, textStatus, errorThrown);
205 			var errorProps = eval('(' + XMLHttpRequest.responseText + ')');
206 			errorCallback(errorProps['status_code'], errorProps['type'], errorProps['explanation']);
207 		};
208 	}; 
209 	
210 	//check with wich type of json the request needs to be made
211 	if(FS.RequestCreator.useJson == true){
212 		//send a json XMLHttpRequest 
213 		$.ajax({
214 			url: uri,
215 			dataType: 'json',
216 			data: dataParams,
217 			success: succesCallback,
218 			error: newErrorCallback,
219 			type: 'GET'
220 		});
221 		
222 	} else{
223 		
224 		$.ajax({
225 			url: uri,
226 			dataType: 'jsonp',
227 			data: dataParams,
228 			success: succesCallback,
229 			//error: newErrorCallback,
230 			type: 'GET'
231 		});
232 	}
233 }

239 /**
240  * @class FreesoundObject class is used to create a base object for Sound, SoundCollection, User, Pack, etc.
241  * @constructor 
242  * @param {object} inProperties Contains the properties of the FreesoundObject
243  */
244 FS.FreesoundObject = function(inProperties){
245 	this.object = this;
246 	this.loaded = true;
247 	
248 	//save properties in properties variable
249 	if (inProperties) { this.properties = inProperties;	}
250 	else {throw new Error("No properties passed to constructor of FreesoundObject.")};
251 };

253 /**
254  * Get item of FreesoundObject
255  * @param {string} item Contains the item name.
256  * @returns {object/string} item 
257  */
258 FS.FreesoundObject.prototype.getItem = function(item){
259 	//check if FreesoundObject object properties are loaded
260 	if(this.loaded){
261 		var myString =  "this.properties." + item;
262 		//check if propertie asked for exists
263 		if (eval(myString)) {
264 			return eval(myString);
265 		}
266 		//propertie doesn't exist, throw error
267 		else{throw new SyntaxError( item + " does not exist in " + this)}
268 	}
269 	//properties aren't loaded, throw error
270 	else {throw new Error("Properties of " + this + " are not loaded.")};
271 }

281 /**
282  * Reload the FreesoundObject.
283  * @param {function} succesCallback Will be called when the request succeeds.
284  * @param {fucntion} errorCallback Will be called when the request fails (no XMLHttpRequest passed and specified error with use of jsonp).
285  */
286 FS.FreesoundObject.prototype.update = function(succesCallback, errorCallback){
287 	if(this.properties.ref)
288 	{
289 		//split uri into uri and parameters object
290 		var uriAndParams = FS._splitUri(this.properties.ref);
291 		//save this (sound, collection, pager, that is calling the update method) inside var object	
292 		var object = this;
293 		//create clossure around object, with which properties can be set 
294 		//and return object, to be able to pass it to callback method		
295 		var setProps = new function(props){
296 		return function(props){ object.properties = props; return object} }
297 		
298 		//call createGetReq method, use above setProps method in callback method to be able to save properties
299 		FS.RequestCreator.createGetReq(uriAndParams[0], function(inProperties){
300 			var obj = setProps(inProperties);
301 			if($.isFunction(succesCallback)){succesCallback(obj); };
302 		}, errorCallback, uriAndParams[1]);	
303 		this.properties = false; 	
304 	}else{ throw new Error("properties of object does not exist, unable to update because no ref. is present");};
305 }

348 /**
349  * Get analysis of Sound object and pass these to the passed callback method 
350  * @param {int} showAll Contains 0 or 1. 0 -> get recommended descriptors, if 1 -> get all descriptors
351  * @param {string} filter Contains a string that contain the path to the wanted descriptor (or gorup of descriptors).
352  * @param {function} succesCallback Will be called when the request succeeds.
353  * @param {function} errorCallback Will be called when the request fails
354  * (no XMLHttpRequest and specified error passed to errorCallback with use of jsonp)
355  */
356 FS.Sound.prototype.getAnalysis = function(showAll, filter, succesCallback, errorCallback){
357 	
358 	//if showAll is undefined -> set to 0
359 	if (!showAll) { showAll = 0;};
360 	//place showAll in object and save to params variable for request	
361 	var params = {all: showAll};
362 	
363 	var uri = FS._URI_SOUND_ANALYSIS;
364 	
365 	//if filter is undefined -> set to array with one empty string
366 	if (!filter) { 	filter = [""];}
367 	
368 	
369 	//create request
370 	FS.RequestCreator.createGetReq(
371 		FS._uri(uri, [this.properties['id'], filter]), 
372 		function(analysis){ 
373 			succesCallback(analysis)
374 		}, errorCallback, params
375 	);

680 		//call succesCallback method

695 /*=============================== Freesound ==============================*/
696 /*======================================================================*/
697 /**
698  * @class Freesound class is the API of the Freesound Javascript Library
699  * @constructor
700  * @param {String} aKey Contains the api key that is going to be used.
701  * @returns {Object} { sound, user, pack, soundCollection }. 
702  */
703 Freesound = function(aKey, useJson){
704 	
705 	//check if there already exist a object of Freesound class (singleton)
706 	if(!Freesound.object)
707 	{
708 		if (aKey) {
709 			
710 			//set use of json
711 			FS.RequestCreator.setUseJson(useJson);
712 				
713 			//set api key
714 			FS.FreesoundData.setApiKey(aKey); 
715 			
716 			//getSound function to call FS.Sound.getSound
717 			var getSound = function(sId, succesCallback, errorCallback)
718 			{	FS.Sound.getSound(sId, succesCallback, errorCallback); };
719 			
720 			//getUser function to call FS.User.getUser
721 			var getUser = function(uId, succesCallback, errorCallback)
722 			{	FS.User.getUser(uId, succesCallback, errorCallback); };
723 			
724 			//getPack function to call FS.Pack.getPack
725 			var getPack = function(pId, succesCallback, errorCallback)
726 			{	FS.Pack.getPack(pId, succesCallback, errorCallback); };
727 			
728 			
729 			//functions that return sound collections
730 			var getSoundsFromQuery = function(query, page, filter, sort, succesCallback, errorCallback)
731 			{	FS.SoundCollection.getSoundsFromQuery(query, page, filter, sort, succesCallback, errorCallback); };
732 			
733 			var getSoundsFromUser = function(uName, succesCallback, errorCallback)
734 			{	FS.SoundCollection.getSoundsFromUser(uName, succesCallback, errorCallback); };
735 			
736 			var getSoundsFromPack = function(pId, succesCallback, errorCallback)
737 			{	FS.SoundCollection.getSoundsFromPack(pId, succesCallback, errorCallback); };
738 			
739 			var getSimilarSoundsFromSound = function(sId, succesCallback, errorCallback)
740 			{	FS.SoundCollection.getSimilarSoundsFromSound(sId, succesCallback, errorCallback); };
741 			
742 			//create Freesound object, containing public variables and methods of Freesound Library						
743 			var object = {
744 				
745 				getApiKey: FS.FreesoundData.getApiKey,
746 				getBaseUri: FS.FreesoundData.getBaseUri,
747 				
748 				getSound: getSound,
749 				
750 				getUser: getUser,
751 				
752 				getPack: getPack,
753 				
754 				getSoundsFromQuery: getSoundsFromQuery,
755 				getSoundsFromUser: getSoundsFromUser,
756 				getSoundsFromPack: getSoundsFromPack,
757 				getSimilarSoundsFromSound: getSimilarSoundsFromSound,
758 				
759 				setTimeout: FS.RequestCreator.setTimeout
760 			};
761 		}
762 		//no correct api key passed -> throw error
763 		else {throw "you didn't pass a api_key to constructor Freesound"; }
764 	}
765 	
766 	
767 	//return object of Freesound class
768 	return object;	
769 }
